create table localizacao
(
	id serial primary key,
	rua varchar(100),
	bairro varchar(100),
	cep varchar(100),
	numero int 
);

create table buffet
(
	id serial primary key,
	fornecedor varchar(100),
	valorTotal real,
	qtdSalgados int,
	qtdDoces int,
	valorBolo real
);

create table fotografo
(
	id serial primary key,
	horaChegada varchar(20),
	horaSaida varchar(20),
	preco real,
	email varchar(100),
	telefone varchar(30),
	linkParaRedeSocial varchar(100)
);

create table cerimonia
(
	id serial primary key,
	horaInicio varchar(10),
	horaFim varchar(10),
	dataCerimonia varchar(20),
	
	id_localizacao int references localizacao(id),
	id_buffet int references buffet(id),
	id_fotografo int references fotografo(id)
);

create table presente
(
	id serial primary key,
	nome varchar(50),
	categoria varchar(50),
	descricao varchar(50),	
	ondeEncontrar varchar(50),
	id_cerimonia int references cerimonia(id)
);

create table usuario
(
	id serial primary key,
	nome varchar(100),
	email varchar(100),
	senha varchar(100),
	telefone varchar(30),
	id_cerimonia int references cerimonia(id)
);

create table convidado
(
	id serial primary key,
	nome varchar(100),
	email varchar(100),
	telefone varchar(30),
	id_cerimonia int references cerimonia(id)
);
